            const timer = window.setInterval(() => {
                if (!text) return;
                frame = (frame + 1) % 3;
                text.textContent = `LOADING${'.'.repeat(frame + 1)}`;
            }, 500);

            window.__vibeBootLoader = {
                hide() {
                    window.clearInterval(timer);
                    if (!overlay) return;
                    overlay.classList.add('is-hidden');
                    window.setTimeout(() => overlay.remove(), 300);
                },
                showError() {
                    window.clearInterval(timer);
                    if (text) text.textContent = 'LOAD ERROR';
                },
            };
        })();
